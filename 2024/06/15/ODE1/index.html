$$

\frac{d(det\phi)}{dt} = \sum_{k = 1}^{n}\begin{vmatrix}
\phi_{11} & \phi_{12} & \frac{d\phi_{1k}}{dt}… & \phi_{1n} \
\phi_{21}& \phi_{22} & \frac{d\phi_{1k}}{dt}… & \phi_{2n} \
\vdots &&&\vdots\
\phi_{n1}& \phi_{n2} &\frac{d\phi_{1k}}{dt}…&\phi_{nn} \
\end{vmatrix}\=
\sum_{k = 1}^{n}\begin{vmatrix}
\phi_{11} & \phi_{12} & \sum_{j=1}^{n}a_{1j}\phi_{jk}… & \phi_{1n} \
\phi_{21}& \phi_{22} &\sum_{j=1}^{n}a_{2j}\phi_{jk}… & \phi_{2n} \
\vdots &&&\vdots\
\phi_{n1}& \phi_{n2} &\sum_{j=1}^{n}a_{nj}\phi_{jk}…&\phi_{nn} \
\end{vmatrix}\=
\sum_{k = 1}^{n}\begin{vmatrix}
\phi_{11} & \phi_{12} & \sum_{j=1}^{n}a_{1j}\phi_{jk}… & \phi_{1n} \
\phi_{21}& \phi_{22} &\sum_{j=1}^{n}a_{2j}\phi_{jk}… & \phi_{2n} \
\vdots &&&\vdots\
\phi_{n1}& \phi_{n2} &\sum_{j=1}^{n}a_{nj}\phi_{jk}…&\phi_{nn} \
\end{vmatrix}\=
\sum_{i = 1}^{n}\begin{vmatrix}
a_{11} & a_{12} & \frac{da_{1k}}{dt}… & a_{1n} \
a_{21}& a_{22} & \frac{da_{1k}}{dt}… & a_{2n} \
\vdots &&&\vdots\
a_{n1}& a_{n2} &\frac{da_{1k}}{dt}…&a_{nn} \
\end{vmatrix}
$$

\vdots&&&\vdots\
\phi_{n1}&\phi_{n2}&…&\phi_{nn}
\end{vmatrix}\=
\sum_{k = 1}^{n}\begin{vmatrix}\

\phi_{11}&\phi_{12}&…&\phi_{1n}\
\vdots&&&\vdots\
a_{kk}\phi_{k1}& a_{kk}\phi_{k2}&…& a_{kk}\phi_{kn}\
\vdots&&&\vdots\
\phi_{n1}&\phi_{n2}&…&\phi_{nn}
\end{vmatrix}
\=\sum_{k = 1}^{n}a_{kk}det(\phi)\
thus \ we\ have\ \det(\phi) = Ce^{\int_{t_0}^{t}{\sum_{k = 1}^{n}a_{kk}(s)ds}}
